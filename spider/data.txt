                lock = false
                // 获取dom对象当前的偏移量：
                var location = domObj.offsetLeft
                console.log(location, target)
                // 步数：
                var step = 0
                clearInterval(timer)
                var timer = setInterval(() => {
                  step = (target - location) / 10
                  step = step > 0 ? Math.ceil(step) : Math.floor(step)
                  location += step
                  domObj.style.left = location + 'px'
                  if (step === 0) {
                    domObj.style.left = target + 'px'
                    clearInterval(timer)
                    lock = true
                  }
                }, 10)
              }
